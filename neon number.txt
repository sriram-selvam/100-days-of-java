public class Main
{
    public static int neonOrNot(int n)
    {
        int sq=n*n,num=0,sum=0;
        while(sq>0)
        {
            num=sq%10;
            sum +=num;
            sq /=10;
        }
        return sum;
    }
    public static void main(String args[])
    {
            int n=23;
            int result=neonOrNot(n);
            if(result==n)
             System.out.print("The given number is Neon number");
            else
            System.out.print("The given number is not a Neon number");
        
    }
}
