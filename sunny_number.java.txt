import java.util.Scanner;
public class Main
{
    public static int  sunnyOrNot(int s)
    {
        s=s+1;
        int i=1;
        while(i<s)
        {
            for(int j=1;j<s;j++)
            {
                if(j==i && j*i==s)
                return 1;
            }
            i++;
        }
        return 0;
    }
    public  static void main(String args[])
    {
        Scanner obj=new Scanner(System.in);
        int s=obj.nextInt();
        int result=sunnyOrNot(s);
     if(result==1)
     {
         System.out.println("its a sunny number");
     }
     else{
         
         System.out.println("its not a sunny number");
     }
        
    }
}
